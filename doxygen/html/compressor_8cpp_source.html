132 // Amount compression:
133 // * If the amount is 0, output 0
134 // * first, divide the amount (in base units) by the largest power of 10 possible; call the exponent e (e is max 9)
135 // * if e<9, the last digit of the resulting number cannot be 0; store it as d, and drop it (divide by 10)
136 // * call the result n
137 // * output 1 + 10*(9*n + d - 1) + e
138 // * if e==9, we only know the resulting number is not zero, so output 1 + 10*(n - 1) + 9
139 // (this is decodable, as d is in [1-9] and e is in [0-9])

141 uint64_t CTxOutCompressor::CompressAmount(uint64_t n)
142 {
143  if (n == 0)
144  return 0;
145  int e = 0;
146  while (((n % 10) == 0) && e < 9) {
147  n /= 10;
148  e++;
149  }
150  if (e < 9) {
151  int d = (n % 10);
152  assert(d >= 1 && d <= 9);
153  n /= 10;
154  return 1 + (n*9 + d - 1)*10 + e;
155  } else {
156  return 1 + (n - 1)*10 + 9;
157  }
158 }

160 uint64_t CTxOutCompressor::DecompressAmount(uint64_t x)
161 {
162  // x = 0 OR x = 1+10*(9*n + d - 1) + e OR x = 1+10*(n - 1) + 9
163  if (x == 0)
164  return 0;
165  x--;
166  // x = 10*(9*n + d - 1) + e
167  int e = x % 10;
168  x /= 10;
169  uint64_t n = 0;
170  if (e < 9) {
171  // x = 9*n + d - 1
172  int d = (x % 9) + 1;
173  x /= 9;
174  // x = n
175  n = x*10 + d;
176  } else {
177  n = x+1;
178  }
179  while (e) {
180  n *= 10;
181  e--;
182  }
183  return n;
184 }
