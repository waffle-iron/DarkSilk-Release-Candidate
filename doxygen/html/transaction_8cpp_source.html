30 double CTransaction::ComputePriority(double dPriorityInputs, unsigned int nTxSize) const
31 {
32  // In order to avoid disincentivizing cleaning up the UTXO set we don't count
33  // the constant overhead for each txin and up to 110 bytes of scriptSig (which
34  // is enough to cover a compressed pubkey p2sh redemption) for priority.
35  // Providing any more cleanup incentive than making additional inputs free would
36  // risk encouraging people to create junk outputs to redeem later.
37  if (nTxSize == 0)
38  nTxSize = ::GetSerializeSize(*this, SER_NETWORK, PROTOCOL_VERSION);
39  BOOST_FOREACH(const CTxIn& txin, vin)
40  {
41  unsigned int offset = 41U + std::min(110U, (unsigned int)txin.scriptSig.size());
42  if (nTxSize > offset)
43  nTxSize -= offset;
44  }
45  if (nTxSize == 0) return 0.0;
46  return dPriorityInputs / nTxSize;
47 }

49 bool CTransaction::CheckTransaction(CValidationState &state)
50 {
51  // Basic checks that don't depend on any context
52  if (vin.empty())
53  return DoS(10, error("CTransaction::CheckTransaction() : vin empty"));
54  if (vout.empty())
55  return DoS(10, error("CTransaction::CheckTransaction() : vout empty"));
56  // Size limits
57  if (::GetSerializeSize(*this, SER_NETWORK, PROTOCOL_VERSION) > MAX_BLOCK_SIZE)
58  return DoS(100, error("CTransaction::CheckTransaction() : size limits failed"));
59 
60  // Check for negative or overflow output values
61  CAmount nValueOut = 0;
62  for (unsigned int i = 0; i < vout.size(); i++)
63  {
64  const CTxOut& txout = vout[i];
65  if (txout.IsEmpty() && !IsCoinBase() && !IsCoinStake())
66  return DoS(100, error("CTransaction::CheckTransaction() : txout empty for user transaction"));
67  if (txout.nValue < 0)
68  return DoS(100, error("CTransaction::CheckTransaction() : txout.nValue negative"));
69  if (txout.nValue > MAX_MONEY)
70  return DoS(100, error("CTransaction::CheckTransaction() : txout.nValue too high"));
71  nValueOut += txout.nValue;
72  if (!MoneyRange(nValueOut))
73  return DoS(100, error("CTransaction::CheckTransaction() : txout total out of range"));
74  }
75 
76  // Check for duplicate inputs
77  set<COutPoint> vInOutPoints;
78  BOOST_FOREACH(const CTxIn& txin, vin)
79  {
80  if (vInOutPoints.count(txin.prevout))
81  return false;
82  vInOutPoints.insert(txin.prevout);
83  }
84 
85  if (IsCoinBase())
86  {
87  if (vin[0].scriptSig.size() < 2 || vin[0].scriptSig.size() > 100)
88  return DoS(100, error("CTransaction::CheckTransaction() : coinbase script size is invalid"));
89  }
90  else
91  {
92  BOOST_FOREACH(const CTxIn& txin, vin)
93  if (txin.prevout.IsNull())
94  return DoS(10, error("CTransaction::CheckTransaction() : prevout is null"));
95  }
96 
97  return true;
98 }
