127 void finalize(const argon2_context *context, argon2_instance_t *instance) {
128  if (context != NULL && instance != NULL) {
129  block blockhash;
130  uint32_t l;
131 
132  copy_block(&blockhash, instance->memory + instance->lane_length - 1);
133 
134  /* XOR the last blocks */
135  for (l = 1; l < instance->lanes; ++l) {
136  uint32_t last_block_in_lane =
137  l * instance->lane_length + (instance->lane_length - 1);
138  xor_block(&blockhash, instance->memory + last_block_in_lane);
139  }
140 
141  /* Hash the result */
142  {
143  uint8_t blockhash_bytes[ARGON2_BLOCK_SIZE];
144  store_block(blockhash_bytes, &blockhash);
145  blake2b_long(context->out, context->outlen, blockhash_bytes,
146  ARGON2_BLOCK_SIZE);
147  secure_wipe_memory(blockhash.v,
148  ARGON2_BLOCK_SIZE); /* clear blockhash */
149  secure_wipe_memory(blockhash_bytes,
150  ARGON2_BLOCK_SIZE); /* clear blockhash_bytes */
151  }
152 
153 #ifdef GENKAT
154  print_tag(context->out, context->outlen);
155 #endif
156 
157  /* Clear memory */
158  clear_memory(instance, context->flags & ARGON2_FLAG_CLEAR_PASSWORD);
159 
160  /* Deallocate the memory */
161  if (NULL != context->free_cbk) {
162  context->free_cbk((uint8_t *)instance->memory,
163  instance->memory_blocks * sizeof(block));
164  } else {
165  free_memory(instance->memory);
166  }
167  }

170 uint32_t index_alpha(const argon2_instance_t *instance,
171  const argon2_position_t *position, uint32_t pseudo_rand,
172  int same_lane) {
173  /*
174  * Pass 0:
175  * This lane : all already finished segments plus already constructed
176  * blocks in this segment
177  * Other lanes : all already finished segments
178  * Pass 1+:
179  * This lane : (SYNC_POINTS - 1) last segments plus already constructed
180  * blocks in this segment
181  * Other lanes : (SYNC_POINTS - 1) last segments
182  */
183  uint32_t reference_area_size;
184  uint64_t relative_position;
185  uint32_t start_position, absolute_position;
186 
187  if (0 == position->pass) {
188  /* First pass */
189  if (0 == position->slice) {
190  /* First slice */
191  reference_area_size =
192  position->index - 1; /* all but the previous */
193  } else {
194  if (same_lane) {
195  /* The same lane => add current segment */
196  reference_area_size =
197  position->slice * instance->segment_length +
198  position->index - 1;
199  } else {
200  reference_area_size =
201  position->slice * instance->segment_length +
202  ((position->index == 0) ? (-1) : 0);
203  }
204  }
205  } else {
206  /* Second pass */
207  if (same_lane) {
208  reference_area_size = instance->lane_length -
209  instance->segment_length + position->index -
210  1;
211  } else {
212  reference_area_size = instance->lane_length -
213  instance->segment_length +
214  ((position->index == 0) ? (-1) : 0);
215  }
216  }
217 
218  /* 1.2.4. Mapping pseudo_rand to 0..<reference_area_size-1> and produce
219  * relative position */
220  relative_position = pseudo_rand;
221  relative_position = relative_position * relative_position >> 32;
222  relative_position = reference_area_size - 1 -
223  (reference_area_size * relative_position >> 32);
224 
225  /* 1.2.5 Computing starting position */
226  start_position = 0;
227 
228  if (0 != position->pass) {
229  start_position = (position->slice == ARGON2_SYNC_POINTS - 1)
230  ? 0
231  : (position->slice + 1) * instance->segment_length;
232  }
233 
234  /* 1.2.6. Computing absolute position */
235  absolute_position = (start_position + relative_position) %
236  instance->lane_length; /* absolute position */
237  return absolute_position;
238 }

252 int fill_memory_blocks(argon2_instance_t *instance) {
253  uint32_t r, s;
254  argon2_thread_handle_t *thread = NULL;
255  argon2_thread_data *thr_data = NULL;
256 
257  if (instance == NULL || instance->lanes == 0) {
258  return ARGON2_THREAD_FAIL;
259  }
260 
261  /* 1. Allocating space for threads */
262  thread = calloc(instance->lanes, sizeof(argon2_thread_handle_t));
263  if (thread == NULL) {
264  return ARGON2_MEMORY_ALLOCATION_ERROR;
265  }
266 
267  thr_data = calloc(instance->lanes, sizeof(argon2_thread_data));
268  if (thr_data == NULL) {
269  free(thread);
270  return ARGON2_MEMORY_ALLOCATION_ERROR;
271  }
272 
273  for (r = 0; r < instance->passes; ++r) {
274  for (s = 0; s < ARGON2_SYNC_POINTS; ++s) {
275  int rc;
276  uint32_t l;
277 
278  /* 2. Calling threads */
279  for (l = 0; l < instance->lanes; ++l) {
280  argon2_position_t position;
281 
282  /* 2.1 Join a thread if limit is exceeded */
283  if (l >= instance->threads) {
284  rc = argon2_thread_join(thread[l - instance->threads]);
285  if (rc) {
286  return ARGON2_THREAD_FAIL;
287  }
288  }
289 
290  /* 2.2 Create thread */
291  position.pass = r;
292  position.lane = l;
293  position.slice = (uint8_t)s;
294  position.index = 0;
295  thr_data[l].instance_ptr =
296  instance; /* preparing the thread input */
297  memcpy(&(thr_data[l].pos), &position,
298  sizeof(argon2_position_t));
299  rc = argon2_thread_create(&thread[l], &fill_segment_thr,
300  (void *)&thr_data[l]);
301  if (rc) {
302  return ARGON2_THREAD_FAIL;
303  }
304 
305  /* fill_segment(instance, position); */
306  /*Non-thread equivalent of the lines above */
307  }
308 
309  /* 3. Joining remaining threads */
310  for (l = instance->lanes - instance->threads; l < instance->lanes;
311  ++l) {
312  rc = argon2_thread_join(thread[l]);
313  if (rc) {
314  return ARGON2_THREAD_FAIL;
315  }
316  }
317  }
318 
319 #ifdef GENKAT
320  internal_kat(instance, r); /* Print all memory blocks */
321 #endif
322  }
323 
324  if (thread != NULL) {
325  free(thread);
326  }
327  if (thr_data != NULL) {
328  free(thr_data);
329  }
330  return ARGON2_OK;
331 }

462 void fill_first_blocks(uint8_t *blockhash, const argon2_instance_t *instance) {
463  uint32_t l;
464  /* Make the first and second block in each lane as G(H0||i||0) or
465  G(H0||i||1) */
466  uint8_t blockhash_bytes[ARGON2_BLOCK_SIZE];
467  for (l = 0; l < instance->lanes; ++l) {
468 
469  store32(blockhash + ARGON2_PREHASH_DIGEST_LENGTH, 0);
470  store32(blockhash + ARGON2_PREHASH_DIGEST_LENGTH + 4, l);
471  blake2b_long(blockhash_bytes, ARGON2_BLOCK_SIZE, blockhash,
472  ARGON2_PREHASH_SEED_LENGTH);
473  load_block(&instance->memory[l * instance->lane_length + 0],
474  blockhash_bytes);
475 
476  store32(blockhash + ARGON2_PREHASH_DIGEST_LENGTH, 1);
477  blake2b_long(blockhash_bytes, ARGON2_BLOCK_SIZE, blockhash,
478  ARGON2_PREHASH_SEED_LENGTH);
479  load_block(&instance->memory[l * instance->lane_length + 1],
480  blockhash_bytes);
481  }
482  secure_wipe_memory(blockhash_bytes, ARGON2_BLOCK_SIZE);
483 }

559 int initialize(argon2_instance_t *instance, argon2_context *context) {
560  uint8_t blockhash[ARGON2_PREHASH_SEED_LENGTH];
561  int result = ARGON2_OK;
562 
563  if (instance == NULL || context == NULL)
564  return ARGON2_INCORRECT_PARAMETER;
565 
566  /* 1. Memory allocation */
567 
568  if (NULL != context->allocate_cbk) {
569  uint8_t *p;
570  result = context->allocate_cbk(&p, instance->memory_blocks *
571  ARGON2_BLOCK_SIZE);
572  if (ARGON2_OK != result) {
573  return result;
574  }
575  memcpy(&(instance->memory), p, sizeof(instance->memory));
576  } else {
577  result = allocate_memory(&(instance->memory), instance->memory_blocks);
578  if (ARGON2_OK != result) {
579  return result;
580  }
581  }
582 
583  /* 2. Initial hashing */
584  /* H_0 + 8 extra bytes to produce the first blocks */
585  /* uint8_t blockhash[ARGON2_PREHASH_SEED_LENGTH]; */
586  /* Hashing all inputs */
587  initial_hash(blockhash, context, instance->type);
588  /* Zeroing 8 extra bytes */
589  secure_wipe_memory(blockhash + ARGON2_PREHASH_DIGEST_LENGTH,
590  ARGON2_PREHASH_SEED_LENGTH -
591  ARGON2_PREHASH_DIGEST_LENGTH);
592 
593 #ifdef GENKAT
594  initial_kat(blockhash, context, instance->type);
595 #endif
596 
597  /* 3. Creating first blocks, we always have at least two blocks in a slice
598  */
599  fill_first_blocks(blockhash, instance);
600  /* Clearing the hash */
601  secure_wipe_memory(blockhash, ARGON2_PREHASH_SEED_LENGTH);
602 
603  return ARGON2_OK;
604 }
