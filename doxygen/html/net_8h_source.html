503  void AskFor(const CInv& inv)
504  {
505  // We're using mapAskFor as a priority queue,
506  // the key is the earliest time the request can be sent
507  int64_t& nRequestTime = mapAlreadyAskedFor[inv];
508  LogPrint("net", "askfor %s %d (%s)\n", inv.ToString(), nRequestTime, DateTimeStrFormat("%H:%M:%S", nRequestTime/1000000));
509 
510  // Make sure not to reuse time indexes to keep things in the same order
511  int64_t nNow = (GetTime() - 1) * 1000000;
512  static int64_t nLastTime;
513  ++nLastTime;
514  nNow = std::max(nNow, nLastTime);
515  nLastTime = nNow;
516 
517  // Each retry is 2 minutes after the last
518  nRequestTime = std::max(nRequestTime + 2 * 60 * 1000000, nNow);
519  mapAskFor.insert(std::make_pair(nRequestTime, inv));
520  }

543  // TODO: Document the precondition of this function. Is cs_vSend locked?
544  void EndMessage() UNLOCK_FUNCTION(cs_vSend)
545  {
546  if (mapArgs.count("-dropmessagestest") && GetRand(atoi(mapArgs["-dropmessagestest"])) == 0)
547  {
548  LogPrint("net", "dropmessages DROPPING SEND MESSAGE\n");
549  AbortMessage();
550  return;
551  }
552 
553  if (ssSend.size() == 0)
554  return;
555 
556  // Set the size
557  unsigned int nSize = ssSend.size() - CMessageHeader::HEADER_SIZE;
558  memcpy((char*)&ssSend[CMessageHeader::MESSAGE_SIZE_OFFSET], &nSize, sizeof(nSize));
559 
560  // Set the checksum
561  uint256 hash = Hash(ssSend.begin() + CMessageHeader::HEADER_SIZE, ssSend.end());
562  unsigned int nChecksum = 0;
563  memcpy(&nChecksum, &hash, sizeof(nChecksum));
564  assert(ssSend.size () >= CMessageHeader::CHECKSUM_OFFSET + sizeof(nChecksum));
565  memcpy((char*)&ssSend[CMessageHeader::CHECKSUM_OFFSET], &nChecksum, sizeof(nChecksum));
566 
567  LogPrint("net", "(%d bytes)\n", nSize);
568 
569  std::deque<CSerializeData>::iterator it = vSendMsg.insert(vSendMsg.end(), CSerializeData());
570  ssSend.GetAndClear(*it);
571  nSendSize += (*it).size();
572 
573  // If write queue empty, attempt "optimistic write"
574  if (it == vSendMsg.begin())
575  SocketSendData(this);
576 
577  LEAVE_CRITICAL_SECTION(cs_vSend);
578  }
